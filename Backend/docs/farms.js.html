/**
 * The `farm` document represents a farm.
 *
 * @name farm
 * @title Farm
 * @type {document}
 *
 * @field {string} name The name of the farm.
 * @field {string} location The location of the farm.
 * @field {string} size The size of the farm.
 * @field {text} description A description of the farm.
 * @field {string} userId The ID of the user who owns the farm.
 * @field {postedBy} postedBy The user who posted the farm.
 * @field {array} memberOf An array of the groups that the farm is a member of.
 * @field {array} animals An array of the animals that live on the farm.
 * @field {array} crops An array of the crops that are grown on the farm.
 * @field {array} equipment An array of the equipment that is used on the farm.
 * @field {array} pests An array of the pests that are found on the farm.
 * @field {datetime} establishedDate The date and time that the farm was established.
 *
 * @option {string} dateFormat The date format for the `establishedDate` field.
 * @option {string} timeFormat The time format for the `establishedDate` field.
 * @option {number} timeStep The time step in minutes for the `establishedDate` field.
 * @option {function} max A function that returns the maximum value for the `establishedDate` field.
 */

export default {
  name: 'farm',
  title: 'Farm',
  type: 'document',
  fields: [
    {
      name: 'name',
      title: 'Name',
      type: 'string',
    },
    {
      name: 'location',
      title: 'Location',
      type: 'string',
    },
    {
      name: 'size',
      title: 'Size',
      type: 'string',
    },
    {
      name: 'description',
      title: 'Description',
      type: 'text',
    },
    {
      name: 'userId',
      title: 'UserID',
      type: 'string',
    },
    {
      name: 'postedBy',
      title: 'PostedBy',
      type: 'postedBy',
    },
    {
      name: 'memberOf',
      title: 'MemberOf',
      type: 'array',
      of: [{type: 'memberOf'}],
    },
    {
      name: 'animals',
      title: 'Animals',
      type: 'array',
      of: [{type: 'animalchoice'}],
    },
    {
      name: 'crops',
      title: 'Crops',
      type: 'array',
      of: [{type: 'cropChoice'}],
    },
    {
      name: 'equipment',
      title: 'Equipment',
      type: 'array',
      of: [{type: 'equipmentChoice'}],
    },
    {
      name: 'pests',
      title: 'Pests',
      type: 'array',
      of: [{type: 'pestsChoice'}],
    },
    {
      name: 'establishedDate',
      title: 'EstablishedDate',
      type: 'datetime',
      options: {
        dateFormat: 'YYYY-MM-DD', // set the date format
        timeFormat: 'HH:mm', // set the time format
        timeStep: 15, // set the time step in minutes
        // set the max value to the current date
        max: () => new Date().toISOString().split('T')[0],
      },
    },
  ],
}
